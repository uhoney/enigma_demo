20void Controller::handle_key_press_event(Vector2& mouse_position)
21{
22 if (this->keyboard.is_key_pressed(mouse_position, IsMouseButtonPressed(MOUSE_LEFT_BUTTON)))
23 {
24 this->play_random_key_sfx();
25
26 char temp_char{ this->keyboard.get_pressed_key()->get_label() };
27
28 this->handle_rotor_turnovers();
29
30 for (auto& rotor : this->rotors)
31 {
32 temp_char = rotor.pass_through(temp_char);
33 }
34
35 temp_char = this->reflector.swap_character(temp_char);
36
37 for (auto rotor = this->rotors.rbegin(); rotor != this->rotors.rend(); ++rotor)
38 {
39 temp_char = rotor->reverse_pass_through(temp_char);
40 }
41
42 this->lampboard.turn_on_lamp(temp_char);
43 }
44
45 if (this->keyboard.is_key_released(IsMouseButtonReleased(MOUSE_LEFT_BUTTON)))
46 {
47 this->lampboard.reset_lamps();
48 }

131void Controller::handle_rotor_press_event(Vector2& mouse_position, bool& mouse_pressed, bool& mouse_released)
132{
133
134 for (auto& rotor : this->rotors)
135 {
136 if (rotor.is_button_pressed(mouse_position, mouse_pressed))
137 {
138 enigma::RotorButton* p_pressed_button = rotor.get_pressed_button();
139 if (p_pressed_button == &rotor.button_up)
140 {
141 this->play_random_key_sfx();
142 rotor.turn_rotor();
143 }
144 else if (p_pressed_button == &rotor.button_down)
145 {
146 this->play_random_key_sfx();
147 rotor.turn_back_rotor();
148 }
149 }
150
151 // just assign nullptr if released ==> single click behavior, no hover over multiple clicks
152 rotor.check_if_button_released(mouse_released);
